# 模板模式
​	模板模式：定义一个操作中的算法框架，而将一些步骤延迟到子类中，使得子类可以不改变一个算法的结构即可重定义该算法的某些特定步骤。

public abstract class AbstractAjhzCounter implements AjHuizCounter {

    /**
     * 庭审公开详情页面列别信息统一处理工具
     * @param config
     * @param r1
     * @param result
     */
    public void handleTsgk(ZbCountConfig config, Map<String, List<Map<String, Object>>> r1, List<Map<String, Object>> result) {
        if (CollectionUtils.isEmpty(result)) {
            Map<String, Object> map = new LinkedHashMap<String, Object>();
            List<Map<String, Object>> list = new ArrayList<Map<String, Object>>();
            map.put("案号", "-");
            map.put("承办人", "-");
            map.put("合议庭成员", "-");
            map.put("法官助理", "-");
            map.put("书记员", "-");
            map.put("结案日期", "-");
            list.add(map);
            r1.put(config.getZbmc(), list);
        } else {
            for (Map<String, Object> re : result) {
                if (null != re.get("ajbs")) {
                    //查询案件关联合议庭成员信息，并处理封装进数据中
                    String[] spcy = spzzcyService.getFgzlSjyHytcySpz(String.valueOf(re.get("ajbs")));
                    re.remove("ajbs");
                    Map<String, Integer> ryzcMap = new HashMap<>(3);
                    ryzcMap.put("hyt", NO_HYTCY);
                    ryzcMap.put("fgzls", NO_FGZL);
                    ryzcMap.put("sjys", NO_SJY);
                    for (Map.Entry<String, Integer> entry : ryzcMap.entrySet()) {
                        if (StringUtils.isNotBlank(spcy[entry.getValue()])) {
                            re.put(entry.getKey(), spcy[entry.getValue()]);
                        }
                    }
                }

                re.put("案号", re.get("ah"));
                re.remove("ah");
                AjhzQueryUtil.clkxx(re, "承办人", "cbr");
                AjhzQueryUtil.clkxx(re, "合议庭成员", "hyt");
                AjhzQueryUtil.clkxx(re, "法官助理", "fgzls");
                AjhzQueryUtil.clkxx(re, "书记员", "sjys");
                re.put("结案日期", re.get("jarq"));
                re.remove("jarq");
            }
            r1.put(config.getZbmc(), result);
        }
    }

}

@Component
public class NTsgkQtcyAjHuizCounter extends AbstractAjhzCounter {

  @Override
  public BigDecimal count(TKpglKpjh kpjh, String bkpr, TKpglKptx kptx) {
    String sql = "SELECT count(DISTINCT(c_ah)) as ajs from db_gjyjda.t_spsj_aj_xzsfgk where n_sfgk=1 and c_hytcy like ? and c_spz != ? and d_jarq BETWEEN ? and ?";
    Object[] objects = new Object[] { "%" + bkpr + "%", bkpr, kpjh.getDKpzqKsrq(), kpjh.getDKpzqJsrq() };
    Map<String, Object> result = jdbcTemplate.queryForMap(sql, objects);
    BigDecimal value = AjhzQueryUtil.buildDcimal(result.get("ajs"));
    return value;
  }

  @Override
  public Map<String, List<Map<String, Object>>> list(TKpglKpjh kpjh, String bkpr, ZbCountConfig config, TKpglKptx kptx, String zbbh) {
    String sql = "SELECT gk.c_ah as ah,gk.n_cbr as cbr,c_hytcy as hyt,to_char(gk.d_jarq,'YYYY-MM-DD') as jarq,gk.c_fgzls as fgzls,gk.c_sjys as sjys from db_gjyjda.t_spsj_aj_xzsfgk gk where gk.n_sfgk=1 and gk.c_hytcy like ? and gk.c_spz != ?and gk.d_jarq BETWEEN ? and ?";
    Object[] objects = new Object[] { "%" + bkpr + "%", bkpr, kpjh.getDKpzqKsrq(), kpjh.getDKpzqJsrq() };
    List<Map<String, Object>> result = jdbcTemplate.queryForList(sql, objects);
    Map<String, List<Map<String, Object>>> r1 = new LinkedHashMap<>();
    handleTsgk(config, r1, result);
    return r1;
  }

}
